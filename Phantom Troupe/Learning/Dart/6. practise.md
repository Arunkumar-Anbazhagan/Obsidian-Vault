void main() {
  double amount = 96;
  double taxRate = 4;
  bool isPremium = true;
  double balance=96;
  totalAmount(amount : amount , taxRate : taxRate , isPremium : isPremium,balance : balance);
  
}

//verify whether the customer is premium or not
double totalAmount({
  required double amount,
  required bool isPremium,
  required double taxRate,
,required double balance}) {
  if (isPremium == true) {
    balanceCheck(balance: balance, amount: amount);
    return amount;
  } else {
    amount = taxCalculation(
      amount: amount,
      taxRate: taxRate
    ); //calculating the tax for the premium users
    balanceCheck(balance : balance, amount : amount);
    return amount;
  }
}

//tax calculation
double taxCalculation({required double amount, required double taxRate}) {
  amount += taxRate;
  return amount;
}

//balance vs the amount needs to be withdrawn
double balanceCheck({required double balance, required double amount}) {
  if (balance >= amount) {
    //balance after withdrawn
    balance -= amount;
    print("Transaction Successful and your remaining balance is $balance");
  } else {
    print("Insufficient Balance");
  }
  return amount;
}
